    cwd=`pwd`
    # Clone sources
    git clone https://github.com/OPENAIRINTERFACE/openair-hss.git
    git clone https://github.com/OPENAIRINTERFACE/openair-mme.git
    git clone https://github.com/OPENAIRINTERFACE/openair-spgwc.git
    git clone https://github.com/OPENAIRINTERFACE/openair-spgwu-tiny.git
  
    # Installing and Building HSS
    cd $cwd/openair-hss
    ./scripts/build_hss_rel14 --check-installed-software --force
    ./scripts/build_hss_rel14 --clean --build-type Release
    
    # Installing and Building MME
    cd $cwd/openair-mme
    ./scripts/build_mme --check-installed-software --force
    ./scripts/build_mme --clean --build-type Release --Verbose
    
    # Installing and Building SPGW-C
    cd $cwd/openair-spgwc
    ./build/scripts/build_spgwc --install-deps --force
    ./build/scripts/build_spgwc --clean --build-type Release --jobs --Verbose

    # Installing and Building SPGW-U-TINY
    cd $cwd/openair-spgwu-tiny
    ./build/scripts/build_spgwu --install-deps --force
    ./build/scripts/build_spgwu --clean --build-type Release --jobs --Verbose
